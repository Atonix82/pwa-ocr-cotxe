<body>

  <h1>Captura o selecciona el tauler del cotxe</h1>


  <!-- Preview de vídeo per fer la foto en directe -->

  <video id="video" autoplay playsinline width="320" height="240"></video>

  <br>

  <button id="capture">Capturar imatge amb la càmera</button>

  <br><br>

  <!-- Botó per seleccionar una imatge de la fototeca -->

  <button id="selectFromLibrary">Selecciona des de la fototeca</button>

  <!-- Element input hidden per a la selecció de fitxer -->

  <input type="file" id="fileInput" accept="image/*" style="display:none" />


  <!-- Canvas per processar la imatge (ocult, servirà per convertir la imatge en DataURL) -->

  <canvas id="canvas" width="320" height="240" style="display:none;"></canvas>


  <!-- Carrega Tesseract.js per l'OCR -->

  <script src="https://cdn.jsdelivr.net/npm/tesseract.js@v4.0.2/dist/tesseract.min.js"></script>

  <script>

    const video = document.getElementById('video');

    const canvas = document.getElementById('canvas');

    const captureButton = document.getElementById('capture');

    const fileInput = document.getElementById('fileInput');

    const selectButton = document.getElementById('selectFromLibrary');

    const ctx = canvas.getContext('2d');


    // Inicia la càmera

    async function startCamera() {

      try {

        const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } });

        video.srcObject = stream;

      } catch (error) {

        console.error('Error al accedir a la càmera:', error);

        alert('No s\'ha pogut accedir a la càmera.');

      }

    }

    startCamera();


    // Captura la imatge del vídeo i processa-la

    captureButton.addEventListener('click', () => {

      ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

      const imageData = canvas.toDataURL('image/png');

      processImage(imageData);

    });


    // Al clicar el botó de selecció, obre el selector de fitxers

    selectButton.addEventListener('click', () => {

      fileInput.click();

    });


    // Quan l'usuari tria una imatge des de la fototeca

    fileInput.addEventListener('change', (event) => {

      const file = event.target.files[0];

      if (!file) return;

      const reader = new FileReader();

      reader.onload = function(e) {

        const imageData = e.target.result; // Data URL de la imatge seleccionada

        processImage(imageData);

      };

      reader.readAsDataURL(file);

    });


    // Processa la imatge amb Tesseract.js (OCR)

    async function processImage(imageData) {

      console.log("Processant la imatge...");

      try {

        const { data: { text } } = await Tesseract.recognize(imageData, 'eng', {

          logger: m => console.log(m)

        });

        console.log("Resultat OCR:", text);

        parseOCRResult(text);

      } catch (error) {

        console.error('Error durant l\'OCR:', error);

      }

    }


    // Processa el text extret per identificar les dades esperades

    function parseOCRResult(text) {

      // Divideix el text en línies i elimina espais innecessaris

      const lines = text.split('\n').map(line => line.trim()).filter(line => line.length > 0);

      console.log("Línies detectades:", lines);


      // Exemple simple d'extracció amb expressions regulars

      const data = {};

      lines.forEach(line => {

        if (/^\d{2}:\d{2}$/.test(line)) {

          data.tempsRecorregut = line;

        } else if (/^\d+(\.\d+)?\s?km$/i.test(line)) {

          data.distanciaRecorregut = line;

        } else if (/^\d+(\.\d+)?\s?L\/100km$/i.test(line)) {

          data.consumRecorregut = line;

        } else if (/^ODO\s*\d+/.test(line)) {

          data.odometre = line;

        }

      });

      console.log("Dades extretes:", data);


      // Afegir data i posició GPS

      data.data = new Date().toISOString();

      getGPSPosition().then(position => {

        data.posicio = position;

        saveEntry(data);

      });

    }


    // Obté la posició GPS si està disponible

    function getGPSPosition() {

      return new Promise((resolve, reject) => {

        if (navigator.geolocation) {

          navigator.geolocation.getCurrentPosition(pos => {

            resolve({ latitude: pos.coords.latitude, longitude: pos.coords.longitude });

          }, error => {

            console.warn("GPS no disponible:", error);

            resolve(null);

          });

        } else {

          resolve(null);

        }

      });

    }


    // Guarda l'entrada en localStorage en format JSON

    function saveEntry(entry) {

      let entries = JSON.parse(localStorage.getItem('vehicleEntries') || '[]');

      entries.push(entry);

      localStorage.setItem('vehicleEntries', JSON.stringify(entries));

      alert('Entrada guardada correctament!');

      console.log("Entrades actuals:", entries);

    }

  </script>
